001package zemeckis;
002
003import javax.annotation.Nonnull;
004import org.realityforge.braincheck.BrainCheckConfig;
005
006/**
007 * Provide access to global configuration settings.
008 */
009public final class Zemeckis
010{
011  private Zemeckis()
012  {
013  }
014
015  /**
016   * Return true if user should pass names into API methods, false if should pass null.
017   *
018   * @return true if user should pass names into API methods, false if should pass null.
019   */
020  public static boolean areNamesEnabled()
021  {
022    return ZemeckisConfig.areNamesEnabled();
023  }
024
025  /**
026   * Return true if uncaught error handlers are enabled.
027   *
028   * @return true if uncaught error handlers are enabled, false otherwise.
029   */
030  public static boolean areUncaughtErrorHandlersEnabled()
031  {
032    return ZemeckisConfig.areUncaughtErrorHandlersEnabled();
033  }
034
035  /**
036   * Return true if invariants will be checked.
037   *
038   * @return true if invariants will be checked.
039   */
040  public static boolean shouldCheckInvariants()
041  {
042    return BrainCheckConfig.checkInvariants();
043  }
044
045  /**
046   * Return true if apiInvariants will be checked.
047   *
048   * @return true if apiInvariants will be checked.
049   */
050  public static boolean shouldCheckApiInvariants()
051  {
052    return BrainCheckConfig.checkApiInvariants();
053  }
054
055  /**
056   * Return true if active tasks will be purged if the scheduler is still running after the maximum number of rounds.
057   *
058   * @return true if active tasks will be purged if the scheduler is still running after the maximum number of rounds.
059   */
060  public static boolean purgeTasksWhenRunawayDetected()
061  {
062    return ZemeckisConfig.purgeTasksWhenRunawayDetected();
063  }
064
065  /**
066   * Add error handler to the list of error handlers called.
067   * The handler should not already be in the list. This method should NOT be called if
068   * {@link #areUncaughtErrorHandlersEnabled()} returns false.
069   *
070   * @param handler the error handler.
071   */
072  public static void addUncaughtErrorHandler( @Nonnull final UncaughtErrorHandler handler )
073  {
074    UncaughtErrorHandlerSupport.get().addUncaughtErrorHandler( handler );
075  }
076
077  /**
078   * Remove error handler from list of existing error handlers.
079   * The handler should already be in the list. This method should NOT be called if
080   * {@link #areUncaughtErrorHandlersEnabled()} returns false.
081   *
082   * @param handler the error handler.
083   */
084  public static void removeUncaughtErrorHandler( @Nonnull final UncaughtErrorHandler handler )
085  {
086    UncaughtErrorHandlerSupport.get().removeUncaughtErrorHandler( handler );
087  }
088
089  /**
090   * Report an uncaught error in stream.
091   *
092   * @param error the error.
093   */
094  public static void reportUncaughtError( @Nonnull final Throwable error )
095  {
096    if ( areUncaughtErrorHandlersEnabled() )
097    {
098      UncaughtErrorHandlerSupport.get().onUncaughtError( error );
099    }
100  }
101}
