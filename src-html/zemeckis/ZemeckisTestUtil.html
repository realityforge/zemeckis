001package zemeckis;
002
003import java.lang.reflect.Field;
004import javax.annotation.Nonnull;
005import javax.annotation.Nullable;
006import org.jetbrains.annotations.TestOnly;
007
008/**
009 * Utility class for interacting with Zemeckis config settings in tests.
010 */
011@GwtIncompatible
012@TestOnly
013public final class ZemeckisTestUtil
014{
015  private ZemeckisTestUtil()
016  {
017  }
018
019  /**
020   * Interface to intercept log messages emitted by runtime.
021   */
022  public interface Logger
023  {
024    void log( @Nonnull String message, @Nullable Throwable throwable );
025  }
026
027  /**
028   * Reset the state of zemeckis config to either production or development state.
029   *
030   * @param productionMode true to set it to production mode configuration, false to set it to development mode config.
031   */
032  @SuppressWarnings( "ConstantConditions" )
033  public static void resetConfig( final boolean productionMode )
034  {
035    if ( ZemeckisConfig.isProductionMode() )
036    {
037      /*
038       * This should really never happen but if it does add assertion (so code stops in debugger) or
039       * failing that throw an exception.
040       */
041      assert !ZemeckisConfig.isProductionMode();
042      throw new IllegalStateException( "Unable to reset config as Zemeckis is in production mode" );
043    }
044
045    if ( productionMode )
046    {
047      disableNames();
048      disableUncaughtErrorHandlers();
049    }
050    else
051    {
052      enableNames();
053      enableUncaughtErrorHandlers();
054    }
055    purgeTasksWhenRunawayDetected();
056    resetState();
057  }
058
059  /**
060   * Reset the state of Zemeckis.
061   * This occasionally needs to be invoked after changing configuration settings in tests.
062   */
063  private static void resetState()
064  {
065    setLogger( null );
066    TemporalScheduler.reset();
067    UncaughtErrorHandlerSupport.reset();
068    VirtualProcessorUnitsHolder.reset();
069  }
070
071  /**
072   * Specify logger to use to capture logging in tests
073   *
074   * @param logger the logger.
075   */
076  public static void setLogger( @Nullable final Logger logger )
077  {
078    /*
079     * This should really never happen but if it does add assertion (so code stops in debugger).
080     */
081    assert !ZemeckisConfig.isProductionMode();
082    if ( ZemeckisConfig.isDevelopmentMode() )
083    {
084      final ZemeckisLogger.ProxyLogger proxyLogger = (ZemeckisLogger.ProxyLogger) ZemeckisLogger.getLogger();
085      proxyLogger.setLogger( null == logger ? null : logger::log );
086    }
087  }
088
089  /**
090   * Set `zemeckis.enable_names` setting to true.
091   */
092  public static void enableNames()
093  {
094    setEnableNames( true );
095  }
096
097  /**
098   * Set `zemeckis.enable_names` setting to false.
099   */
100  public static void disableNames()
101  {
102    setEnableNames( false );
103  }
104
105  /**
106   * Configure the `zemeckis.enable_names` setting.
107   *
108   * @param value the setting.
109   */
110  private static void setEnableNames( final boolean value )
111  {
112    setConstant( "ENABLE_NAMES", value );
113  }
114
115  /**
116   * Set `zemeckis.purge_tasks_when_runaway_detected` setting to true.
117   */
118  public static void purgeTasksWhenRunawayDetected()
119  {
120    setPurgeTasksWhenRunawayDetected( true );
121  }
122
123  /**
124   * Set `zemeckis.purge_tasks_when_runaway_detected` setting to false.
125   */
126  public static void noPurgeTasksWhenRunawayDetected()
127  {
128    setPurgeTasksWhenRunawayDetected( false );
129  }
130
131  /**
132   * Configure the `zemeckis.purge_tasks_when_runaway_detected` setting.
133   *
134   * @param value the setting.
135   */
136  private static void setPurgeTasksWhenRunawayDetected( final boolean value )
137  {
138    setConstant( "PURGE_ON_RUNAWAY", value );
139  }
140
141  /**
142   * Set `zemeckis.enable_uncaught_error_handlers` setting to true.
143   */
144  public static void enableUncaughtErrorHandlers()
145  {
146    setEnableUncaughtErrorHandlers( true );
147  }
148
149  /**
150   * Set `zemeckis.enable_uncaught_error_handlers` setting to false.
151   */
152  public static void disableUncaughtErrorHandlers()
153  {
154    setEnableUncaughtErrorHandlers( false );
155  }
156
157  /**
158   * Configure the `zemeckis.enable_uncaught_error_handlers` setting.
159   *
160   * @param value the setting.
161   */
162  private static void setEnableUncaughtErrorHandlers( final boolean value )
163  {
164    setConstant( "UNCAUGHT_ERROR_HANDLERS", value );
165  }
166
167  /**
168   * Set the specified field name on ZemeckisConfig.
169   */
170  private static void setConstant( @Nonnull final String fieldName, final boolean value )
171  {
172    if ( ZemeckisConfig.isDevelopmentMode() )
173    {
174      try
175      {
176        final Field field = ZemeckisConfig.class.getDeclaredField( fieldName );
177        field.setAccessible( true );
178        field.set( null, value );
179      }
180      catch ( final NoSuchFieldException | IllegalAccessException e )
181      {
182        throw new IllegalStateException( "Unable to change constant " + fieldName, e );
183      }
184    }
185    else
186    {
187      throw new AssertionError( "Unable to change constant " + fieldName + " as Zemeckis is in production mode" );
188    }
189  }
190}
