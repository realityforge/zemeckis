001package zemeckis;
002
003import java.util.Objects;
004import javax.annotation.Nonnull;
005import javax.annotation.Nullable;
006import org.jetbrains.annotations.TestOnly;
007import static org.realityforge.braincheck.Guards.*;
008
009/**
010 * Processing unit responsible for executing tasks.
011 */
012public final class VirtualProcessorUnit
013{
014  /**
015   * A human consumable name for node. It should be non-null if {@link Zemeckis#areNamesEnabled()} returns
016   * true and <tt>null</tt> otherwise.
017   */
018  @Nullable
019  private final String _name;
020  /**
021   * The executor responsible for selecting and invoking tasks.
022   */
023  @Nonnull
024  private final Executor _executor;
025
026  /**
027   * Create the processor unit.
028   *
029   * @param executor the associated task executor.
030   */
031  VirtualProcessorUnit( @Nullable final String name, @Nonnull final Executor executor )
032  {
033    if ( Zemeckis.shouldCheckApiInvariants() )
034    {
035      apiInvariant( () -> Zemeckis.areNamesEnabled() || null == name,
036                    () -> "Zemeckis-0002: VirtualProcessorUnit passed a name '" + name + "' but " +
037                          "Zemeckis.areNamesEnabled() is false" );
038    }
039    _name = Zemeckis.areNamesEnabled() ? Objects.requireNonNull( name ) : null;
040    _executor = Objects.requireNonNull( executor );
041    _executor.init( activationFn -> VirtualProcessorUnitsHolder.activate( this, activationFn ) );
042  }
043
044  /**
045   * Return the name of the VirtualProcessorUnit.
046   * This method should NOT be invoked unless {@link Zemeckis#areNamesEnabled()} returns true and will throw an
047   * exception if invariant checking is enabled.
048   *
049   * @return the name of the VirtualProcessorUnit.
050   */
051  @Nonnull
052  public String getName()
053  {
054    if ( Zemeckis.shouldCheckApiInvariants() )
055    {
056      apiInvariant( Zemeckis::areNamesEnabled,
057                    () -> "Zemeckis-0003: VirtualProcessorUnit.getName() invoked when Zemeckis.areNamesEnabled() is false" );
058    }
059    assert null != _name;
060    return _name;
061  }
062
063  /**
064   * Queue task for execution and enable the executor for activation if necessary.
065   * The task must not be already queued.
066   *
067   * @param task the task.
068   */
069  public void queue( @Nonnull final Runnable task )
070  {
071    getExecutor().queue( task );
072  }
073
074  @Nonnull
075  Executor getExecutor()
076  {
077    return _executor;
078  }
079
080  @Nonnull
081  @Override
082  public String toString()
083  {
084    if ( Zemeckis.areNamesEnabled() )
085    {
086      return getName();
087    }
088    else
089    {
090      return super.toString();
091    }
092  }
093
094  /**
095   * Interface via which the {@link VirtualProcessorUnit} executes tasks.
096   * The executor is responsible for activating the underlying {@link VirtualProcessorUnit} when required.
097   * Each time the {@link VirtualProcessorUnit} is activated it will use callback to talk to executor
098   * and the executor is responsible for selecting and executing tasks until it decides to return control
099   * to the {@link VirtualProcessorUnit}.
100   */
101  interface Executor
102  {
103    /**
104     * Initialize the executor passing in the context associated with the underlying {@link VirtualProcessorUnit}.
105     *
106     * @param context the context represent the associated {@link VirtualProcessorUnit}.
107     */
108    void init( @Nonnull Context context );
109
110    /**
111     * Reset state of executor.
112     * This is only used from within the test framework.
113     */
114    @TestOnly
115    void reset();
116
117    /**
118     * Queue task for execution and enable the executor for activation if necessary.
119     * The task must not be already queued.
120     *
121     * @param task the task.
122     */
123    void queue( @Nonnull Runnable task );
124
125    /**
126     * Queue task for execution next. The executor is not activated. The task must not be already queued.
127     *
128     * @param task the task.
129     */
130    void queueNext( @Nonnull Runnable task );
131
132    /**
133     * Activate the executor.
134     */
135    void activate();
136  }
137
138  @FunctionalInterface
139  interface ActivationFn
140  {
141    /**
142     * Callback method invoked by {@link Context#activate(ActivationFn)} to process tasks.
143     */
144    void invoke();
145  }
146
147  /**
148   * Interface representing {@link VirtualProcessorUnit} passed to {@link Executor} during initialization.
149   * This interface is designed to allow the {@link Executor} to activate the {@link VirtualProcessorUnit}
150   * when it needs to execute tasks.
151   */
152  interface Context
153  {
154    /**
155     * Activate the associated {@link VirtualProcessorUnit}.
156     * This method MUST only be called if there is no {@link VirtualProcessorUnit} unit currently activated.
157     * The activation will set the "current" VPU for the duration of the activation and invoke the
158     * {@link ActivationFn#invoke()} function passed into the method.
159     *
160     * @param activationFn the function passed to process tasks.
161     */
162    void activate( @Nonnull ActivationFn activationFn );
163  }
164}
